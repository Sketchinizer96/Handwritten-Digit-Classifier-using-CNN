#!/usr/bin/env bash
# setup-code-review-graph.sh
# Creates the full .claude/ directory structure for code-review-graph agent.
# Run this from the root of any project.

set -euo pipefail

BOLD="\033[1m"
GREEN="\033[32m"
CYAN="\033[36m"
DIM="\033[2m"
RESET="\033[0m"

ok()  { echo -e "${GREEN}✓${RESET}  $*"; }
log() { echo -e "${DIM}create${RESET}  $*"; }
h1()  { echo -e "\n${BOLD}${CYAN}$*${RESET}"; }

ROOT=".claude"

# ── guard ─────────────────────────────────────────────────────────────────────

if [[ -d "$ROOT" ]]; then
  echo -e "\n⚠  .claude/ already exists in this directory."
  read -rp "   Overwrite? [y/N] " yn
  [[ "$yn" =~ ^[Yy]$ ]] || { echo "Aborted."; exit 0; }
fi

# ── directories ───────────────────────────────────────────────────────────────

h1 "Creating directory structure"
mkdir -p "$ROOT/hooks"
mkdir -p "$ROOT/skills/daily-review"
mkdir -p "$ROOT/skills/pr-gate"
mkdir -p "$ROOT/skills/onboard"
mkdir -p "$ROOT/skills/multi-repo"
ok "Directories ready"

# ── CLAUDE.md ─────────────────────────────────────────────────────────────────

h1 "Writing CLAUDE.md"
log "$ROOT/CLAUDE.md"
cat > "$ROOT/CLAUDE.md" << 'HEREDOC'
# code-review-graph Agent

You are a code review agent powered by `code-review-graph`. Use the CLI proactively — don't wait to be asked. Skills in `.claude/skills/` are available for reusable workflows; invoke them automatically when the task matches.

## CLI Reference

```
code-review-graph build              # Parse codebase, build graph (run once per session)
code-review-graph update             # Incremental update for changed files only
code-review-graph watch              # Auto-update on file changes (background)
code-review-graph status             # Graph stats: nodes, edges, files
code-review-graph detect-changes     # Risk-scored change impact analysis
code-review-graph visualize          # Interactive HTML graph (D3.js)
code-review-graph visualize --format graphml   # Export for Gephi/yEd
code-review-graph visualize --format svg       # Static SVG
code-review-graph visualize --format obsidian  # Obsidian vault with wikilinks
code-review-graph visualize --format cypher    # Neo4j Cypher
code-review-graph wiki               # Generate markdown wiki from communities
code-review-graph eval               # Run evaluation benchmarks
code-review-graph eval --all         # Benchmarks across all repos
code-review-graph register <path>    # Add repo to multi-repo registry
code-review-graph unregister <id>    # Remove repo from registry
code-review-graph repos              # List registered repositories
```

## Slash Commands

```
/code-review-graph:build-graph       # Build or rebuild the code graph
/code-review-graph:review-delta      # Review changes since last commit
/code-review-graph:review-pr         # Full PR review with blast-radius analysis
```

## Skills (invoke automatically when task matches)

| Skill | Trigger |
|---|---|
| `daily-review` | User asks for code review, diff review, or "what changed" |
| `pr-gate` | User mentions PR, merge, or pull request review |
| `onboard` | User mentions onboarding, new developer, or architecture overview |
| `multi-repo` | User mentions multiple repos or cross-repo analysis |

## Defaults & Rules

- Always run `build` before any analysis if no graph exists or if `status` shows stale data
- Prefer `update` over `build` when only a few files changed — it's faster
- Run `watch` in background during active development sessions
- When running `detect-changes`, always include risk scores in your response
- For visualizations, default to HTML unless the user specifies a format
- Never skip `eval` when assessing graph accuracy on a new codebase
- Always report `status` at the start of a review session

## Output Format

For every review session:

```
## Graph status
<output of `status`>

## Risk summary
<top changed files with risk scores from `detect-changes`>

## Analysis
<narrative review>

## Recommendations
<prioritised action items>
```
HEREDOC
ok "CLAUDE.md"

# ── settings.json ─────────────────────────────────────────────────────────────

h1 "Writing settings.json"
log "$ROOT/settings.json"
cat > "$ROOT/settings.json" << 'HEREDOC'
{
  "hooks": {
    "PostToolUse": [
      {
        "matcher": "Bash",
        "hooks": [
          {
            "type": "command",
            "command": "bash .claude/hooks/post-bash.sh"
          }
        ]
      },
      {
        "matcher": "Write|Edit|MultiEdit",
        "hooks": [
          {
            "type": "command",
            "command": "bash .claude/hooks/post-file-write.sh"
          }
        ]
      }
    ],
    "PreToolUse": [
      {
        "matcher": "Bash",
        "hooks": [
          {
            "type": "command",
            "command": "bash .claude/hooks/pre-bash.sh"
          }
        ]
      }
    ],
    "Stop": [
      {
        "hooks": [
          {
            "type": "command",
            "command": "bash .claude/hooks/session-end.sh"
          }
        ]
      }
    ]
  }
}
HEREDOC
ok "settings.json"

# ── hooks ─────────────────────────────────────────────────────────────────────

h1 "Writing hooks"

log "$ROOT/hooks/pre-bash.sh"
cat > "$ROOT/hooks/pre-bash.sh" << 'HEREDOC'
#!/usr/bin/env bash
# .claude/hooks/pre-bash.sh
# Fires before every Bash tool call.
# Warns when a destructive git operation is about to run so the graph
# can be snapshotted first if needed.

TOOL_INPUT="${CLAUDE_TOOL_INPUT:-}"

is_destructive_git_op() {
  echo "$TOOL_INPUT" | grep -qE "(git reset --hard|git clean|git push --force|git rebase)"
}

graph_exists() {
  code-review-graph status &>/dev/null 2>&1
}

if is_destructive_git_op && graph_exists; then
  echo "[hook:pre-bash] ⚠ Destructive git operation detected."
  echo "[hook:pre-bash] Capturing graph status snapshot before proceeding..."
  code-review-graph status
  echo "[hook:pre-bash] Snapshot complete. Proceeding with command."
fi
HEREDOC
ok "hooks/pre-bash.sh"

log "$ROOT/hooks/post-bash.sh"
cat > "$ROOT/hooks/post-bash.sh" << 'HEREDOC'
#!/usr/bin/env bash
# .claude/hooks/post-bash.sh
# Fires after every Bash tool call.
# Detects git commits/checkouts and triggers a graph update automatically.

TOOL_OUTPUT="${CLAUDE_TOOL_OUTPUT:-}"
TOOL_INPUT="${CLAUDE_TOOL_INPUT:-}"

is_git_op() {
  echo "$TOOL_INPUT" | grep -qE "(git commit|git checkout|git switch|git merge|git rebase|git pull|git reset)"
}

graph_exists() {
  code-review-graph status &>/dev/null 2>&1
}

if is_git_op; then
  echo "[hook:post-bash] Git operation detected — updating graph..."
  if graph_exists; then
    code-review-graph update
    echo "[hook:post-bash] Graph updated."
  else
    echo "[hook:post-bash] No graph found — skipping update (run 'code-review-graph build' first)."
  fi
fi
HEREDOC
ok "hooks/post-bash.sh"

log "$ROOT/hooks/post-file-write.sh"
cat > "$ROOT/hooks/post-file-write.sh" << 'HEREDOC'
#!/usr/bin/env bash
# .claude/hooks/post-file-write.sh
# Fires after every Write / Edit / MultiEdit tool call.
# Runs an incremental graph update so the graph stays current as Claude edits files.

FILE_PATH="${CLAUDE_TOOL_FILE_PATH:-}"

# Skip non-source files (config, markdown, lock files, etc.)
skip_file() {
  echo "$FILE_PATH" | grep -qE "\.(md|json|lock|yaml|yml|toml|txt|env|gitignore)$"
}

graph_exists() {
  code-review-graph status &>/dev/null 2>&1
}

if skip_file; then
  exit 0
fi

if graph_exists; then
  echo "[hook:post-file-write] Source file edited ($FILE_PATH) — running incremental update..."
  code-review-graph update
  echo "[hook:post-file-write] Graph updated."
fi
HEREDOC
ok "hooks/post-file-write.sh"

log "$ROOT/hooks/session-end.sh"
cat > "$ROOT/hooks/session-end.sh" << 'HEREDOC'
#!/usr/bin/env bash
# .claude/hooks/session-end.sh
# Fires when the Claude Code session ends (Stop event).
# Runs detect-changes and saves a fresh visualization as a session summary.

graph_exists() {
  code-review-graph status &>/dev/null 2>&1
}

if ! graph_exists; then
  exit 0
fi

echo "[hook:session-end] Session ending — generating summary..."

echo ""
echo "=== Graph status ==="
code-review-graph status

echo ""
echo "=== Change impact (risk scores) ==="
code-review-graph detect-changes

echo ""
echo "=== Saving visualization ==="
code-review-graph visualize
echo "[hook:session-end] Visualization saved. Open the HTML file to explore changes from this session."
HEREDOC
ok "hooks/session-end.sh"

chmod +x "$ROOT"/hooks/*.sh
ok "Hook permissions set"

# ── skills ────────────────────────────────────────────────────────────────────

h1 "Writing skills"

log "$ROOT/skills/daily-review/SKILL.md"
cat > "$ROOT/skills/daily-review/SKILL.md" << 'HEREDOC'
---
name: daily-review
description: >
  Daily code review workflow. Invoke when the user asks to review changes,
  review a diff, check what changed, or run a routine code review.
allowed-tools: Bash
---

Run a daily code review using code-review-graph.

Steps:
1. Check graph freshness with `code-review-graph status`
2. If graph is stale or missing, run `code-review-graph build`, otherwise run `code-review-graph update`
3. Run `code-review-graph detect-changes` to get risk-scored change impact
4. Run `/code-review-graph:review-delta` for narrative summary of changes since last commit
5. Identify files with the highest risk scores and flag them for priority human review

Output your findings in this structure:

## Graph status
<status output>

## Risk summary
<top changed files with risk scores — highest first>

## Analysis
<narrative: what changed, why it matters, blast radius>

## Recommendations
<prioritised list of files/modules requiring human attention>
HEREDOC
ok "skills/daily-review/SKILL.md"

log "$ROOT/skills/pr-gate/SKILL.md"
cat > "$ROOT/skills/pr-gate/SKILL.md" << 'HEREDOC'
---
name: pr-gate
description: >
  PR merge gate workflow. Invoke when the user mentions a pull request, wants
  to review before merging, or asks for blast-radius analysis on a branch.
allowed-tools: Bash
---

Run a full PR merge gate check using code-review-graph.

Steps:
1. Run `code-review-graph build` to ensure graph reflects the current branch
2. Run `code-review-graph status` — report node/edge counts and file coverage
3. Run `code-review-graph detect-changes` — collect risk scores for all changed files
4. Run `code-review-graph eval` — validate graph accuracy before using it for decisions
5. Run `/code-review-graph:review-pr` for full blast-radius analysis
6. Determine merge recommendation based on risk scores:
   - Any file with critical risk → recommend BLOCK
   - High risk files present → recommend REVIEW before merge
   - Low/medium risk only → recommend APPROVE with notes

Output a structured gate report:

## PR gate report

**Graph status**: <nodes / edges / files>
**Eval score**: <benchmark result>
**Overall risk**: <CRITICAL | HIGH | MEDIUM | LOW>
**Merge recommendation**: <BLOCK | REVIEW | APPROVE>

## Risk breakdown
<table of changed files, risk score, affected modules>

## Blast radius
<modules and files transitively impacted by this PR>

## Required actions
<specific steps before this PR can be merged>
HEREDOC
ok "skills/pr-gate/SKILL.md"

log "$ROOT/skills/onboard/SKILL.md"
cat > "$ROOT/skills/onboard/SKILL.md" << 'HEREDOC'
---
name: onboard
description: >
  Developer onboarding workflow. Invoke when a new developer needs to understand
  the codebase, when the user asks for an architecture overview, or requests
  onboarding materials.
allowed-tools: Bash
---

Generate comprehensive onboarding materials using code-review-graph.

Steps:
1. Run `code-review-graph build` to ensure graph is up to date
2. Run `code-review-graph status` — summarise codebase scale for the new dev
3. Run `code-review-graph wiki` — generate a markdown wiki from code communities
4. Run `code-review-graph visualize` — produce an interactive HTML graph for exploration
5. Run `code-review-graph detect-changes` — identify the most active/volatile areas to be aware of
6. Based on the graph and wiki, produce a written onboarding guide covering:
   - High-level architecture and key modules
   - Entry points (main files, bootstrapping, routing)
   - Hotspots — frequently changed, high-risk files new devs should treat carefully
   - Ownership areas — which communities own which parts of the codebase
   - Local setup steps inferred from the graph (build commands, config files found)
   - Recommended reading order for exploring the codebase

Output:

## Codebase overview
<scale, languages, key modules>

## Architecture map
<module breakdown with responsibilities>

## Key entry points
<files to read first>

## Hotspots & high-risk areas
<files/modules to approach with caution>

## Onboarding checklist
<step-by-step for a new developer's first day>

The HTML visualization has been saved — direct the developer to open it in their browser.
The wiki has been saved — link them to the generated markdown files.
HEREDOC
ok "skills/onboard/SKILL.md"

log "$ROOT/skills/multi-repo/SKILL.md"
cat > "$ROOT/skills/multi-repo/SKILL.md" << 'HEREDOC'
---
name: multi-repo
description: >
  Multi-repo analysis workflow. Invoke when the user mentions multiple
  repositories, cross-repo dependencies, or wants a combined risk report
  across a monorepo or microservices setup.
allowed-tools: Bash
---

Run a cross-repository analysis using code-review-graph.

Steps:
1. Run `code-review-graph repos` — list currently registered repositories
2. For any repos not yet registered, run `code-review-graph register <path>` for each
3. Run `code-review-graph eval --all` — benchmark graph accuracy across all repos
4. For each registered repo, run `code-review-graph detect-changes` to collect per-repo risk scores
5. Aggregate findings into a unified risk report
6. Identify cross-repo blast radius: changes in one repo that could affect others
7. Recommend which repos require attention before any shared deployment

Output:

## Registered repositories
<list from `repos` command>

## Eval results (all repos)
<benchmark scores per repo>

## Per-repo risk summary
<repo name → top risk files → risk score>

## Cross-repo impact
<changes that propagate across repo boundaries>

## Deployment recommendation
<which repos are safe to deploy, which need review>
HEREDOC
ok "skills/multi-repo/SKILL.md"

# ── summary ───────────────────────────────────────────────────────────────────

h1 "Done"
echo -e "Created the following structure:\n"
find "$ROOT" -type f | sort | sed "s|^|  ${DIM}|;s|$|${RESET}|"
echo -e "\n${BOLD}Next steps:${RESET}"
echo "  1. cd into your project root (if not already there)"
echo "  2. Run this script:  bash setup-code-review-graph.sh"
echo "  3. Install the tool: pipx install code-review-graph"
echo "  4. Build the graph:  code-review-graph build"
echo "  5. Open Claude Code — skills and hooks are active immediately"
echo ""
